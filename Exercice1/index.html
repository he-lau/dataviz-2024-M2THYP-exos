Wikidata

Liste des romans pour lesquels une image est disponible.

Requête SPARQL

#defaultView:Timeline
SELECT ?roman ?titreLabel ?date_de_publication ?nombre_de_pages ?authorLabel ?image WHERE {
?roman wdt:P31 wd:Q8261.
SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE],en". }
OPTIONAL { ?roman wdt:P1476 ?titre. }
OPTIONAL { ?roman wdt:P577 ?date_de_publication. }
OPTIONAL { ?roman wdt:P50 ?author. }
OPTIONAL { ?roman wdt:P1104 ?nombre_de_pages. }
OPTIONAL { ?roman wdt:P18 ?image. }
FILTER EXISTS { ?roman wdt:P18 [] }
}
LIMIT 100
